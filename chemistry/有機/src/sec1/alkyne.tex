\subsection{アルキン\:\:\ce{C_{n}H_{2n-2}}\:(Alkyne)}

\begin{enumerate}[label=$\cdot$]
  \item アルカンの名称をane\rightarrow yneにする（アセチレンは慣用名）
  \item 水に不溶
\end{enumerate}
\subsubsection*{アセチレンの製法}
\eeg{
  \ce{CaC2 + 2H2O -> \chemfig{HC~CH} ^ + Ca(OH)2}
}
\subsubsection*{アセチレンの重合}
\begin{enumerate}[label=$\cdot$]
  \item 二量化
  \eeg{
    \ce{2\chemfig{HC~CH} ->C[Cu^{2+}] \chemfig{C(-[3]H)(-[-3]H)=C(-[1]H)(-[-1]C~CH)}}
  }
  \item 三量化
  \eeg{
    \ce{
      3\chemfig{HC~CH} ->C[Fe,$\Delta$] \benzene
    }
  }
\end{enumerate}
